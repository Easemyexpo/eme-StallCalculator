    <script>
        // Global currency settings
        let currentCurrency = 'USD';
        let currencySymbols = {
            'USD': '$', 'EUR': '‚Ç¨', 'GBP': '¬£', 'CAD': 'C$', 'AUD': 'A$',
            'JPY': '¬•', 'CNY': '¬•', 'INR': '‚Çπ', 'SGD': 'S$', 'CHF': 'Fr'
        };

        // Market cost multipliers
        const marketMultipliers = {
            'low': 0.6,
            'medium': 1.0,
            'high': 1.4,
            'premium': 1.8
        };

        // Base costs by market and currency
        const baseCosts = {
            boothRates: {
                // Global rates in USD (will be converted)
                'shell': 120, 'modular': 180, 'custom': 300, 'rental': 220, 'octonorm': 200,
                // India-specific rates in INR per sq meter
                'india': {
                    'shell': 8000, 'modular': 10000, 'custom': 18000, 'rental': 14000, 'octonorm': 12000
                }
            },
            positionMultipliers: {
                'inline': 1.0, 'corner': 1.25, 'island': 1.5, 'peninsula': 1.35
            },
            accommodationRates: {
                'budget': 60, 'business': 140, 'luxury': 280, 'ultra': 450,
                // India-specific rates in INR per night
                'india': {
                    'budget': 2500, 'business': 6000, 'luxury': 12000, 'ultra': 25000
                }
            },
            travelRates: {
                'flight': 0.25, 'train': 0.15, 'car': 0.08, 'bus': 0.05,
                // India rates per km in INR
                'india': {
                    'flight': 8, 'train': 3, 'car': 6, 'bus': 2
                }
            },
            laborRates: {
                'default': 125,
                'india': 800 // INR per hour for Indian market
            },
            drayageRates: {
                'default': 150,
                'india': 120 // INR per 100kg in India
            }
        };

        function updateCurrency() {
            currentCurrency = document.getElementById('currency').value;
            calculateTotals();
        }

        function getBoothSize() {
            const sizeSelect = document.getElementById('booth-size').value;
            if (sizeSelect === 'custom') {
                return parseFloat(document.getElementById('custom-size').value) || 36;
            }
            return parseFloat(sizeSelect);
        }

        function calculateBoothCost() {
            const boothSize = getBoothSize();
            const boothType = document.getElementById('booth-type').value;
            const position = document.getElementById('booth-position').value;
            const marketLevel = document.getElementById('market-level').value;
            
            let baseRate;
            
            // Use India-specific pricing for INR currency
            if (currentCurrency === 'INR') {
                baseRate = baseCosts.boothRates.india[boothType] || 12000; // Default to octonorm rate
            } else {
                baseRate = baseCosts.boothRates[boothType] || 180;
            }
            
            let positionMultiplier = baseCosts.positionMultipliers[position] || 1.0;
            let marketMultiplier = marketMultipliers[marketLevel] || 1.0;
            
            let baseCost = boothSize * baseRate * positionMultiplier;
            
            // Apply market multiplier only for non-INR currencies (India rates are already localized)
            if (currentCurrency !== 'INR') {
                baseCost *= marketMultiplier;
            }
            
            // Add feature costs (scaled appropriately for currency)
            let featureCost = 0;
            let featureMultiplier = currentCurrency === 'INR' ? 80 : 1; // INR feature multiplier
            
            if (document.getElementById('premium-flooring').checked) featureCost += boothSize * 25 * featureMultiplier;
            if (document.getElementById('special-lighting').checked) featureCost += boothSize * 35 * featureMultiplier;
            if (document.getElementById('furniture-rental').checked) featureCost += boothSize * 40 * featureMultiplier;
            if (document.getElementById('storage-space').checked) featureCost += 800 * featureMultiplier;
            if (document.getElementById('meeting-room').checked) featureCost += 1500 * featureMultiplier;
            
            return baseCost + featureCost;
        }

        function calculateTravelCost() {
            const distance = parseFloat(document.getElementById('distance').value) || 0;
            const staffCount = parseInt(document.getElementById('staff-count').value) || 1;
            const accommodationNights = parseInt(document.getElementById('accommodation-nights').value) || 4;
            const accommodationLevel = document.getElementById('accommodation-level').value;
            const travelMethod = document.getElementById('travel-method').value;
            const marketLevel = document.getElementById('market-level').value;
            const eventDuration = parseInt(document.getElementById('event-duration').value) || 3;
            const mealBudget = parseFloat(document.getElementById('meal-budget').value) || 75;
            const localTransport = parseFloat(document.getElementById('local-transport').value) || 200;
            const incidentals = parseFloat(document.getElementById('incidentals').value) || 30;
            
            let totalCost = 0;
            let marketMultiplier = (currentCurrency === 'INR') ? 1.0 : (marketMultipliers[marketLevel] || 1.0);
            
            // Travel costs
            let travelRate;
            if (currentCurrency === 'INR') {
                travelRate = baseCosts.travelRates.india[travelMethod] || 3;
            } else {
                travelRate = baseCosts.travelRates[travelMethod] || 0.15;
            }
            totalCost += distance * travelRate * staffCount * 2; // Round trip
            
            // Accommodation
            let accommodationRate;
            if (currentCurrency === 'INR') {
                accommodationRate = baseCosts.accommodationRates.india[accommodationLevel] || 6000;
            } else {
                accommodationRate = baseCosts.accommodationRates[accommodationLevel] || 140;
            }
            totalCost += accommodationRate * accommodationNights * staffCount;
            
            // Meals and incidentals - scale for currency
            let currencyMultiplier = currentCurrency === 'INR' ? 80 : 1;
            totalCost += (mealBudget + incidentals) * eventDuration * staffCount * currencyMultiplier;
            totalCost += localTransport * currencyMultiplier;
            
            return totalCost * marketMultiplier;
        }

        function calculateShippingCost() {
            const distance = parseFloat(document.getElementById('distance').value) || 0;
            const materialWeight = parseFloat(document.getElementById('material-weight').value) || 0;
            const shippingMethod = document.getElementById('shipping-method').value;
            const marketLevel = document.getElementById('market-level').value;
            
            if (materialWeight === 0) return 0;
            
            let marketMultiplier = marketMultipliers[marketLevel] || 1.0;
            let shippingRates = {
                'standard': 2.5, 'expedited': 4.0, 'freight': 6.5, 'courier': 8.0
            };
            
            let baseCost = materialWeight * shippingRates[shippingMethod];
            let distanceCost = distance * 0.02 * materialWeight;
            
            return (baseCost + distanceCost) * 2 * marketMultiplier; // Round trip
        }

        function calculateDrayageCost() {
            const materialWeight = parseFloat(document.getElementById('material-weight').value) || 0;
            let drayageRate = parseFloat(document.getElementById('drayage-rate').value);
            const shipmentCount = parseInt(document.getElementById('shipment-count').value) || 1;
            const marketLevel = document.getElementById('market-level').value;
            
            if (materialWeight === 0) return 0;
            
            // Use India-specific default rate if not specified
            if (!drayageRate) {
                drayageRate = currentCurrency === 'INR' ? baseCosts.drayageRates.india : baseCosts.drayageRates.default;
            }
            
            let marketMultiplier = (currentCurrency === 'INR') ? 1.0 : (marketMultipliers[marketLevel] || 1.0);
            
            // Calculate CWT (per 100kg units)
            let cwt = Math.ceil(materialWeight / 100);
            let totalDrayage = cwt * drayageRate * shipmentCount * 2; // In and out
            
            // Minimum charge per shipment (scaled for currency)
            let minimumCharge = (currentCurrency === 'INR' ? 15000 : 200) * shipmentCount * 2;
            
            return Math.max(totalDrayage, minimumCharge) * marketMultiplier;
        }

        function calculateLaborCost() {
            let laborRate = parseFloat(document.getElementById('labor-rate').value);
            const laborHours = parseFloat(document.getElementById('labor-hours').value) || 12;
            const setupTiming = document.getElementById('setup-timing').value;
            const unionRequired = document.getElementById('union-required').value;
            const marketLevel = document.getElementById('market-level').value;
            
            // Use India-specific default rate if not specified
            if (!laborRate) {
                laborRate = currentCurrency === 'INR' ? baseCosts.laborRates.india : baseCosts.laborRates.default;
            }
            
            let marketMultiplier = (currentCurrency === 'INR') ? 1.0 : (marketMultipliers[marketLevel] || 1.0);
            let timingMultiplier = 1.0;
            let unionMultiplier = 1.0;
            
            // Timing multipliers
            if (setupTiming === 'overtime') timingMultiplier = 1.5;
            if (setupTiming === 'weekend') timingMultiplier = 2.0;
            
            // Union multipliers (less relevant in India)
            if (unionRequired === 'partial') unionMultiplier = currentCurrency === 'INR' ? 1.1 : 1.2;
            if (unionRequired === 'full') unionMultiplier = currentCurrency === 'INR' ? 1.2 : 1.5;
            
            return laborRate * laborHours * timingMultiplier * unionMultiplier * marketMultiplier;
        }

        function calculateTechnologyCost() {
            const internetPackage = document.getElementById('internet-package').value;
            const electricalPackage = document.getElementById('electrical-package').value;
            const eventDuration = parseInt(document.getElementById('event-duration').value) || 3;
            const marketLevel = document.getElementById('market-level').value;
            
            let cost = 0;
            let marketMultiplier = marketMultipliers[marketLevel] || 1.0;
            
            // Internet costs
            const internetCosts = { 'none': 0, 'basic': 200, 'premium': 500, 'dedicated': 1200 };
            cost += (internetCosts[internetPackage] || 0);
            
            // Electrical costs
            const electricalCosts = { 'basic': 300, 'enhanced': 600, 'heavy': 1000, 'custom': 1500 };
            cost += (electricalCosts[electricalPackage] || 0);
            
            // Additional tech services
            if (document.getElementById('av-rental').checked) cost += 800 * eventDuration;
            if (document.getElementById('led-screens').checked) cost += 1500 * eventDuration;
            if (document.getElementById('sound-system').checked) cost += 400 * eventDuration;
            if (document.getElementById('interactive-tech').checked) cost += 1000 * eventDuration;
            if (document.getElementById('live-streaming').checked) cost += 2000;
            
            return cost * marketMultiplier;
        }

        function calculateMarketingCost() {
            const marketLevel = document.getElementById('market-level').value;
            let cost = 0;
            let marketMultiplier = marketMultipliers[marketLevel] || 1.0;
            
            if (document.getElementById('printed-materials').checked) cost += 800;
            if (document.getElementById('promotional-items').checked) cost += 1500;
            if (document.getElementById('signage-graphics').checked) cost += 1200;
            if (document.getElementById('product-samples').checked) cost += 600;
            if (document.getElementById('digital-materials').checked) cost += 2000;
            
            return cost * marketMultiplier;
        }

        function calculateAdditionalCost() {
            const eventDuration = parseInt(document.getElementById('event-duration').value) || 3;
            const staffCount = parseInt(document.getElementById('staff-count').value) || 1;
            const marketLevel = document.getElementById('market-level').value;
            
            let cost = 0;
            let marketMultiplier = marketMultipliers[marketLevel] || 1.0;
            
            if (document.getElementById('security-service').checked) cost += 150 * eventDuration;
            if (document.getElementById('cleaning-service').checked) cost += 100 * eventDuration;
            if (document.getElementById('technical-support').checked) cost += 200 * eventDuration;
            if (document.getElementById('hospitality-service').checked) cost += 300 * eventDuration;
            if (document.getElementById('lead-capture').checked) cost += 500;
            
            return cost * marketMultiplier;
        }

        function calculateTotals() {
            const boothCost = calculateBoothCost();
            const travelCost = calculateTravelCost();
            const shippingCost = calculateShippingCost();
            const drayageCost = calculateDrayageCost();
            const laborCost = calculateLaborCost();
            const technologyCost = calculateTechnologyCost();
            const marketingCost = calculateMarketingCost();
            const additionalCost = calculateAdditionalCost();
            
            const subtotal = boothCost + travelCost + shippingCost + drayageCost + 
                           laborCost + technologyCost + marketingCost + additionalCost;
            
            const contingencyRate = parseFloat(document.getElementById('contingency-rate').value) || 0.20;
            const contingencyAmount = subtotal * contingencyRate;
            const totalBudget = subtotal + contingencyAmount;
            
            // Update display
            document.getElementById('booth-cost').textContent = formatCurrency(boothCost);
            document.getElementById('travel-cost').textContent = formatCurrency(travelCost);
            document.getElementById('shipping-cost').textContent = formatCurrency(shippingCost);
            document.getElementById('drayage-cost').textContent = formatCurrency(drayageCost);
            document.getElementById('labor-cost').textContent = formatCurrency(laborCost);
            document.getElementById('technology-cost').textContent = formatCurrency(technologyCost);
            document.getElementById('marketing-cost').textContent = formatCurrency(marketingCost);
            document.getElementById('additional-cost').textContent = formatCurrency(additionalCost);
            document.getElementById('subtotal').textContent = formatCurrency(subtotal);
            document.getElementById('contingency-amount').textContent = formatCurrency(contingencyAmount);
            document.getElementById('total-budget').textContent = formatCurrency(totalBudget);
        }

        function formatCurrency(amount) {
            const symbol = currencySymbols[currentCurrency] || '$';
            
            if (currentCurrency === 'JPY' || currentCurrency === 'CNY') {
                return symbol + Math.round(amount).toLocaleString();
            }
            
            return symbol + amount.toLocaleString('en-US', {
                minimumFractionDigits: 0,
                maximumFractionDigits: 0
            });
        }

            if (confirm('Are you sure you want to reset all values?')) {
                // Reset to defaults
                document.getElementById('currency').value = 'USD';
                document.getElementById('market-level').value = 'medium';
                document.getElementById('event-duration').value = '3';

                document.getElementById('local-transport').value = '200';

                document.getElementById('contingency-rate').value = '0.20';
                
                // Reset selects to first option
                const selects = document.querySelectorAll('select:not(#currency):not(#market-level):not(#contingency-rate)');
                selects.forEach(select => select.selectedIndex = 0);

                calculateTotals();
            }

        // Add event listeners
        document.addEventListener('DOMContentLoaded', function() {
            const inputs = document.querySelectorAll('input, select');
            inputs.forEach(input => {
                input.addEventListener('change', function() {
                    if (this.id === 'currency') {
                        updateCurrency();
                    } else {
                        calculateTotals();
                    }
                });
                input.addEventListener('input', calculateTotals);
            });
            
            // Initial calculation
            calculateTotals();
        });
